3   /**
4    * LedED computes the Levenstein edit distance between two words.
5    *
6    * @author smayfiel
7    *
8    */
9   public class LevED {
10  
11    /**
12     * Empty constructor included for completeness.
13     */
14    public LevED() {
15  
16    }
17  
18    /**
19     * Calculates the Levenstein edit distance between two words.
20     *
21     * @param word1
22     *          the first word
23     * @param word2
24     *          the second word
25     * @return the edit distance to get from word1 to word2
26     */
27    public int distance(String word1, String word2) {
28  
29      // Buffer words with spaces
30      word1 = " ".concat(word1);
31      word2 = " ".concat(word2);
32  
33      // Initialize array
34      int numCols = word1.length();
35      int numRows = word2.length();
36      int[][] distArray = new int[numRows][numCols];
37  
38      // Fill in first row and column
39      for (int r = 0; r < numRows; r++) {
40        distArray[r][0] = r;
41      }
42      for (int c = 0; c < numCols; c++) {
43        distArray[0][c] = c;
44      }
45      // Fill in the rest of the array
46      for (int r = 1; r < numRows; r++) {
47        for (int c = 1; c < numCols; c++) {
48          // Check for same letter
49          int diff = 1;
50          if (word1.charAt(c) == word2.charAt(r)) {
51            diff = 0;
52          }
53          distArray[r][c] = minOf3(distArray[r - 1][c] + 1,
54              distArray[r][c - 1] + 1, distArray[r - 1][c - 1] + diff);
55        }
56      }
57      return distArray[numRows - 1][numCols - 1];
58    }
59  
60    /**
61     * Private helper method to compute the minimum of three integers.
62     *
63     * @param x
64     *          the first integer
65     * @param y
66     *          the second integer
67     * @param z
68     *          the third integer
69     * @return the min of x,y, and z
70     */
71    private int minOf3(int x, int y, int z) {
72      if (x < y && x < z) {
73        return x;
74      } else if (y < x && y < z) {
75        return y;
76      } else {
77        return z;
78      }
79    }
80  }
