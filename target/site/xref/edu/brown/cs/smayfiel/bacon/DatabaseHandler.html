10  /**
11   * The DatabaseHandler is responsible for getting a connection to the database
12   * and then giving it to whomever needs it.
13   *
14   * @author smayfiel
15   *
16   */
17  public class DatabaseHandler {
18  
19    private static String db;
20  
21    /**
22     * Constructor takes in a path to the database, supplied on the command line.
23     *
24     * @param db
25     *          the string path to the database
26     * @throws ClassNotFoundException
27     *           if setting up the connection fails
28     */
29    public DatabaseHandler(String db) throws ClassNotFoundException {
30      DatabaseHandler.db = db;
31      Class.forName("org.sqlite.JDBC");
32  
33    }
34  
35    /**
36     * This constructor does not take in a path - it hard codes a path to the
37     * database. This constructor is only used for testing, but would be useful if
38     * you were working in a situation where you were always working with the same
39     * database in the same place, so I left it for future use/
40     *
41     * @throws ClassNotFoundException
42     *           if setting up the connection fails
43     */
44    public DatabaseHandler() throws ClassNotFoundException {
45      DatabaseHandler.db = "/course/cs032/data/bacon/smallBacon.sqlite3";
46      Class.forName("org.sqlite.JDBC");
47  
48    }
49  
50    /**
51     * Method to return the connection to the database so other classes can query
52     * it. This method is static.
53     *
54     * @return the connection to the database
55     */
56    public static Connection getConnection() {
57      try {
58        String urlToDB = "jdbc:sqlite:" + db;
59        Connection conn = DriverManager.getConnection(urlToDB);
60        // Connection conn = DriverManager.getConnection(URL);
61        Statement stat = conn.createStatement();
62        stat.executeUpdate("PRAGMA foreign_keys = ON;");
63        return conn;
64      } catch (SQLException se) {
65        throw new RuntimeException(se);
66      }
67    }
68  
69    /**
70     * This is a specific query that the InputHandler has that is handled here to
71     * minimize the sql scattered around.
72     *
73     * @param name
74     *          the name of an actor to find
75     * @return the id of the actor if it exists, null otherwise
76     */
77    public String getActorIDByName(String name) {
78      Connection conn = DatabaseHandler.getConnection();
79      String id = "";
80      try (PreparedStatement prep = conn
81          .prepareStatement("SELECT id FROM actor WHERE name = ?;")) {
82        prep.setString(1, name);
83        try (ResultSet rs = prep.executeQuery()) {
84          rs.next();
85          id = rs.getString(1);
86        }
87      } catch (SQLException se) {
88        System.out.println("ERROR: No actor ID matches that name");
89        return null;
90      }
91      return id;
92    }
93  
94    /**
95     * The equivalent of getActorByName() for movies. This method is never called
96     * but is included for completion.
97     *
98     * @param name
99     *          the name of the movie
100    * @return the id of the movie if it exists, null otherwise
101    */
102   public String getMovieIDByName(String name) {
103     Connection conn = DatabaseHandler.getConnection();
104     String id = "";
105     try (PreparedStatement prep = conn
106         .prepareStatement("SELECT id FROM film WHERE name = ?;")) {
107       prep.setString(1, name);
108       try (ResultSet rs = prep.executeQuery()) {
109         rs.next();
110         id = rs.getString(1);
111       }
112     } catch (SQLException se) {
113       System.out.println("ERROR: No movie ID matches that name");
114       return null;
115     }
116     return id;
117   }
118 }
