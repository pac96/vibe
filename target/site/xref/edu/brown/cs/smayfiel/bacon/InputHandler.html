9   /**
10   * This class takes the arguments from Main, parses them, calls Dijkstra, and
11   * then prints the results to the terminal.
12   *
13   * @author smayfiel
14   *
15   */
16  public class InputHandler {
17  
18    private List<String> args;
19  
20    /**
21     * Constructor takes in args from Main.
22     *
23     * @param args
24     *          the arguments from Main
25     */
26    public InputHandler(List<String> args) {
27      this.args = args;
28    }
29  
30    /**
31     * Method to create Dijkstra and print path found.
32     *
33     * @return the list of strings to print to the user. These are either of the
34     *         form "A1 -/- A2" or the form "A1 - A2 : M1"
35     */
36    public List<String> runBacon() {
37      DatabaseHandler dbh = null;
38      try {
39        dbh = new DatabaseHandler(args.get(2));
40      } catch (ClassNotFoundException e1) {
41        System.out.println("Database not found");
42        System.exit(1);
43      }
44      // Actor names in string form
45      String firstActor = args.get(0);
46      String lastActor = args.get(1);
47      // Actor IDs
48      String id1 = dbh.getActorIDByName(firstActor);
49      if (id1 == null) {
50        return null;
51      }
52      String id2 = dbh.getActorIDByName(lastActor);
53      // If these are null, the databasehandler printed the error message
54      if (id2 == null) {
55        return null;
56      }
57  
58      // Creating the cache
59      new IDCache();
60  
61      Actor start = new Actor(id1);
62      start.setName(firstActor);
63      IDCache.getCache().put(id1, start);
64  
65      Actor finish = new Actor(id2);
66      finish.setName(lastActor);
67      IDCache.getCache().put(id2, finish);
68  
69      // Creating Dijkstra
70      Dijkstra d2 = new Dijkstra();
71      List<Nodeable> results = d2.findShortestPath(start, finish);
72  
73      // This is the list that will be returned to main
74      List<String> returnList = new ArrayList<>();
75      // No path
76      if (results == null) {
77        returnList.add(firstActor + " -/- " + lastActor);
78      } else {
79        for (int i = 0; i < results.size() - 1; i++) {
80          String a1Name = results.get(i).getName();
81          String a2Name = results.get(i + 1).getName();
82          String mName = results.get(i + 1).getPrevEdge().getData().getName();
83          returnList.add(a1Name + " -> " + a2Name + " : " + mName);
84        }
85      }
86      return returnList;
87    }
88  }
