34      c = new Actor("C");
35      d = new Actor("D");
36      e = new Actor("E");

38      Edge e2 = new Edge(new Movie("2"), a, c);
39      Edge e3 = new Edge(new Movie("3"), b, c);
40      Edge e4 = new Edge(new Movie("2"), b, d);
41      Edge e5 = new Edge(new Movie("3"), b, e);
42      Edge e6 = new Edge(new Movie("1"), c, b);
43      Edge e7 = new Edge(new Movie("4"), c, d);
44      Edge e8 = new Edge(new Movie("5"), c, e);
45      Edge e9 = new Edge(new Movie("1"), e, d);
46      e1.getData().setWeight(4.0);
47      e2.getData().setWeight(2.0);
48      e3.getData().setWeight(3.0);
49      e4.getData().setWeight(2.0);
50      e5.getData().setWeight(3.0);
51      e6.getData().setWeight(1.0);
52      e7.getData().setWeight(4.0);
53      e8.getData().setWeight(5.0);
54      e9.getData().setWeight(1.0);
55      a.setEdges(Arrays.asList(e1, e2));
56      b.setEdges(Arrays.asList(e3, e4, e5));
57      c.setEdges(Arrays.asList(e6, e7, e8));

64    }

66    @Test
67    public void oneNodeTest() {
68      List<Nodeable> results = dij.findShortestPath(a, a);
69      assertTrue(results.size() == 1);
70      assertTrue(results.get(0) == a);
71    }
72  
73    @Test
74    public void twoNodeTest() {
75      List<Nodeable> results = dij.findShortestPath(a, c);
76      assertTrue(results.size() == 2);
77      assertTrue(results.get(0) == a);
78      assertTrue(results.get(1) == c);
79    }
80  
81    @Test
82    public void noPathTest() {
83      Nodeable f = new Actor("F");
84      List<Nodeable> results = dij.findShortestPath(a, f);
85      assertTrue(results == null);
86    }
87  
88    @Test
89    public void longerTest() {
90      List<Nodeable> results = dij.findShortestPath(a, e);
91      assertTrue(results.size() == 4);
92      assertTrue(results.get(0) == a);
93      assertTrue(results.get(1) == c);
94      assertTrue(results.get(2) == b);
95      assertTrue(results.get(3) == e);
96    }
