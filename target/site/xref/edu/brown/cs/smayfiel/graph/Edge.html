3   /**
4    * Edge is a direct graph edge that tracks three things: an object stored in the
5    * Edge of type Edgeable, the Nodeable that this directed edge comes from, and
6    * the Nodeable this edge goes to.
7    *
8    * @author smayfiel
9    *
10   */
11  public class Edge {
12  
13    private Edgeable data;
14    private Nodeable fromNode;
15    private Nodeable toNode;
16  
17    /**
18     * Constructor initializes the ivars to the arguments.
19     *
20     * @param data
21     *          the object stored by this Edge
22     * @param fromA
23     *          the node this edge comes from
24     * @param toA
25     *          the node this edge goes to
26     */
27    public Edge(Edgeable data, Nodeable fromA, Nodeable toA) {
28      this.data = data;
29      fromNode = fromA;
30      toNode = toA;
31    }
32  
33    /**
34     * Gets the object stored in this Edge.
35     *
36     * @return Edgeable oject
37     */
38    public Edgeable getData() {
39      return data;
40    }
41  
42    /**
43     * Gets the Nodeable this Edge comes from.
44     *
45     * @return fromNode
46     */
47    public Nodeable getFromNode() {
48      return fromNode;
49    }
50  
51    /**
52     * Gets the Nodeable this Edge goes to.
53     *
54     * @return toNode
55     */
56    public Nodeable getToNode() {
57      return toNode;
58    }
59  }
