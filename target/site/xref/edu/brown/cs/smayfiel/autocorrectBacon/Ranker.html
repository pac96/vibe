3   import java.util.List;

5   /**
6    * This interface creates the outline of a Ranker class that will get
7    * suggestions and order them according to some rule. Both the BasicRanker and
8    * my SmartRanker implement this interface.
9    *
10   * @author smayfiel
11   *
12   */
13  public interface Ranker {
14  
15    /**
16     * Returns a list of suggestions ranked best to worse based on two input
17     * strings.
18     *
19     * @param s1
20     *          the first string to consider
21     * @param s2
22     *          the second string to consider
23     * @return list of ranked suggestions
24     */
25    List<String> getRankedSuggestions(String s1, String s2);
26  
27    /**
28     * Method to determine which kinds of suggestions to use. This method need not
29     * be called if the user wants the default behavior.
30     *
31     * @param usePrefix
32     *          boolean determining if prefix suggestions are found
33     * @param useLed
34     *          boolean determining if prefix suggestions are found
35     * @param maxEdits
36     *          max number of edits, unused if useLed is false
37     * @param useWhitespace
38     *          boolean determining if prefix suggestions are found
39     */
40    void setOptions(boolean usePrefix, boolean useLed, int maxEdits,
41        boolean useWhitespace);
42  }
