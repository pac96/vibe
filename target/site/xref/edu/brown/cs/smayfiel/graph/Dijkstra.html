6   import java.util.ArrayList;
7   import java.util.List;
8   import java.util.PriorityQueue;

10  /**
11   * Class that performs Dijkstra's Algorithm without ever having the entire graph
12   * in memory. Works generically: any class that implements the Nodeable
13   * interface can be treated as a Node, and Edges store anything of type
14   * Edgeable.
15   *
16   * @author smayfiel
17   *
18   */
19  public class Dijkstra {
20  
21    /**
22     * Empty constructor included for completeness.
23     */
24    public Dijkstra() {
25  
26    }
27  
28    /**
29     * Method that actually runs Dijkstra's algorithm. When the dst node is found,
30     * getPath is called to actually return the shortest path. If no path can be
31     * found, null is returned.
32     *
33     * @param src
34     *          the starting node
35     * @param dst
36     *          the ending node
37     * @return a list of nodes along the path (from src to dst) or null if no path
38     *         can be found
39     */
40    public List<Nodeable> findShortestPath(Nodeable src, Nodeable dst) {
41  
42      // Set up pq
43      PriorityQueue<Nodeable> pq = new PriorityQueue<Nodeable>(
44          new NodeableComparator());
45      // Set up source
46      src.setDist(0);
47      pq.add(src);
48  
49      // Main while loop
50      while (!pq.isEmpty()) {
51        Nodeable current = pq.remove();
52        current.markVisited();
53  
54        // If this is what we are looking for
55        if (current.equals(dst)) {
56          return this.getPath(current);
57        }
58        List<Edge> list = current.getEdges();
59  
60        for (Edge edge : list) {
61          Nodeable next = edge.getToNode();
62  
63          if (!next.isVisited()) {
64            // If this path is shorter, update everything
65            if (current.getDist() + edge.getData().getWeight() < next.getDist()) {
66  
67              next.setDist(current.getDist() + edge.getData().getWeight());
68  
69              next.setPrevNode(current);
70              next.setPrevEdge(edge);
71              if (!next.inPQ()) {
72                pq.add(next);
73                next.markInPQ();
74              } else {
75                pq.remove(next);
76                pq.add(next);
77              }
78            }
79          }
80        }
81      }
82      return null;
83    }
84  
85    /**
86     * Method to reverse the path through the graph, because each Nodeable is
87     * storing a reference to the previous node, but we generally want the path
88     * from start to finish, not finish to start. This method returns to
89     * findShortestPath()
90     *
91     * @param dst
92     *          the last node in the path
93     * @return a list of Nodeables in order from start to finish
94     */
95    private List<Nodeable> getPath(Nodeable dst) {
96      List<Nodeable> returnList = new ArrayList<>();
97      Nodeable current = dst;
98      while (current != null) {
99        // Add each new node to the front of the list
100       returnList.add(0, current);
101       // Update current pointer
102       current = current.getPrevNode();
103     }
104     return returnList;
105   }
106 }
